	LDA strHeaderAddr
	STA pAD_name
	LDA headerAddr
	STA pAD_str
	BSA printAsDeclaration
	LDA newLine

	LDA strFooterAddr
	STA pAD_name
	LDA footerAddr
	STA pAD_str
	BSA printAsDeclaration
	LDA footerAddr
	STA print_str
	BSA print
	BUN opsys


pAD_name: W 1
pAD_str: W 1
pAD_char: W 1
printAsDeclaration: W 1
	LDA pAD_name
	STA print_str
	BSA print
	LDA strDeclStartAddr
	STA print_str
	BSA print
	LDA blankLda
	IOR pAD_str
	STA pAD_lda
pAD_lda: W 1
	IOR zero
	BZE pAD_endLoop
	STA pAD_char
	DIV sixteen
	SUB ten
	AND signBit
	BZE pAD_letter0
	LDA pAD_char
	DIV sixteen
	ADD char0
	BUN pAD_out0
pAD_letter0: LDA pAD_char
	DIV sixteen
	SUB ten
	ADD upperA
pAD_out0: OUT pAD_out0
	LDA pAD_char
	MOD sixteen
	SUB ten
	AND signBit
	BZE pAD_letter1
	LDA pAD_char
	MOD sixteen
	ADD char0
	BUN pAD_out1
pAD_letter1: LDA pAD_char
	MOD sixteen
	SUB ten
	ADD upperA
pAD_out1: OUT pAD_out1
	LDA pAD_lda
	ADD one
	STA pAD_lda
	LDA strDeclNextAddr
	STA print_str
	BSA print
	BUN pAD_lda
pAD_endLoop: LDA strDeclEndAddr
	STA print_str
	BSA print
	BIN printAsDeclaration


print_str: W 1
print: W 1
	LDA blankLda
	IOR print_str
	STA print_lda
print_lda: W 1
	IOR zero
	BZE print_ret
print_out: OUT print_out
	LDA print_lda
	ADD one
	STA print_lda
	BUN print_lda
print_ret: BIN print

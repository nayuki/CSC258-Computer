W 0  # Prints "Hello, world!\n". Standalone program with no dependencies.


msgstr:
	C 'H'
	C 'e'
	C 'l'
	C 'l'
	C 'o'
	C ','
	C ' '
	C 'w'
	C 'o'
	C 'r'
	C 'l'
	C 'd'
	C '!'
	C '\n'
	C '\0'

zero: I 0
one:  I 1
mask: H 00FFFFFF


main:
	load: LDA msgstr
	IOR zero
	BZE opsys
	out: OUT out
	
	LDA load
	ADD one
	AND mask
	STA load
	
	BUN main
